let count = 0;

function cc(card) {
  // Only change code below this line
switch(card){
  case 2: case 3: case 4: case 5: case 6:
    count++;
    break;
  case 'J': case 'K': case 'Q': case 'A': case 10:
    count--;
    break;
}
if(count<=0){
  return count +" Hold";
}else{
  return count +" Bet";
}
  // Only change code above this line
}

cc(2); cc(3); cc(7); cc('K'); cc('A');
